package com.aichatvn.agent.skills

import com.aichatvn.agent.core.AgentKernel.PluginResult
import com.aichatvn.agent.core.plugin.Plugin
import com.aichatvn.agent.core.plugin.PluginAction
import com.aichatvn.agent.core.plugin.PluginParameter
import com.aichatvn.agent.core.plugin.PluginCapabilities
import com.aichatvn.agent.core.plugin.PluginManifest
import com.aichatvn.agent.skills.base.BaseSkill
import com.aichatvn.agent.utils.Logger
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class MyNewSkill @Inject constructor(
    logger: Logger
) : BaseSkill("my_plugin_id", "Tên Plugin Mới", logger), Plugin {

    // ✅ KHAI BÁO MANIFEST ĐẦY ĐỦ: Tuyên bố năng lực và danh sách hành động
    override val manifest = PluginManifest(
        id = id,
        name = name,
        capabilities = PluginCapabilities(
            dashboard = false,    // Đổi thành true nếu muốn vẽ sơ đồ lên Dashboard
            training = true       // Đổi thành true nếu muốn tự đồng bộ trigger intent vào DB
        ),
        routable = true,          // Cho phép AgentKernel phân tuyến trực tiếp
        visibleOnDashboard = false,
        autoGenerateQA = true,
        actions = listOf(
            PluginAction(
                name = "my_action",
                description = "Mô tả chi tiết hành động để AI hiểu và phân loại lệnh",
                examples = listOf("câu lệnh mẫu 1", "câu lệnh mẫu 2"),
                triggerPrefixes = listOf("KíchHoạt:"), // Cú pháp kích hoạt nhanh nếu có
                parameters = listOf(
                    PluginParameter("param_1", "string", "Mô tả tham số 1", required = true)
                )
            )
        )
    )

    // ✅ TIẾP NHẬN THỰC THI NGHIỆP VỤ: Đã được Kernel làm sạch dữ liệu đầu vào
    override suspend fun execute(action: String, params: Map<String, Any>): PluginResult {
        return when (action) {
            "my_action" -> {
                val paramVal = params["param_1"] as? String 
                    ?: return failure("Thiếu tham số bắt buộc 'param_1'")
                
                // Viết mã nghiệp vụ thực tế của bạn ở đây...
                
                success("Đã xử lý thành công giá trị: $paramVal")
            }
            else -> failure("Hành động không được hỗ trợ trong plugin này.")
        }
    }

    override suspend fun initialize() {
        // Khởi tạo các biến, kết nối phần cứng hoặc cấu hình ban đầu
    }

    override suspend fun shutdown() {
        // Dọn dẹp tài nguyên hoặc ngắt kết nối khi hệ thống tắt
    }
}
